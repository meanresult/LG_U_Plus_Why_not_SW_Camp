python
from PIL import Image, ImageDraw
import os

# Load the base image
base_image = Image.open("image.png").convert("RGBA")

# Create 4 variations to simulate waving
frames = []
width, height = base_image.size

# Define hand movement offsets (simulate waving)
offsets = [0, -5, 5, 0]

for offset in offsets:
    frame = base_image.copy()
    draw = ImageDraw.Draw(frame)
    
    # Simulate hand movement by drawing a translucent ellipse over the hand area
    hand_area = (width - 110, height // 2 - 60 + offset, width - 60, height // 2 - 10 + offset)
    draw.ellipse(hand_area, fill=(255, 255, 255, 40))
    
    frames.append(frame)

# Save as animated GIF
gif_path = "/mnt/data/waving_character.gif"
frames[0].save(gif_path, save_all=True, append_images=frames[1:], optimize=False, duration=500, loop=0)

    loop=0

python
import requests
from PIL import Image
from io import BytesIO
import zipfile
import os

# 이미지 URL들
image_urls = [
    "https://cdn.jsdelivr.net/gh/epic-sauce/gif-assets@main/wave1.png",
    "https://cdn.jsdelivr.net/gh/epic-sauce/gif-assets@main/wave2.png",
    "https://cdn.jsdelivr.net/gh/epic-sauce/gif-assets@main/wave3.png",
    "https://cdn.jsdelivr.net/gh/epic-sauce/gif-assets@main/wave1.png",
]

# 이미지 다운로드 및 저장
image_files = []
for idx, url in enumerate(image_urls):
    response = requests.get(url)
    image = Image.open(BytesIO(response.content))
    filename = f"wave_frame_{idx + 1}.png"
    image.save(filename)
    image_files.append(filename)

# ZIP 파일로 압축
zip_filename = "/mnt/data/waving_frames.zip"
with zipfile.ZipFile(zip_filename, "w") as zipf:
    for file in image_files:
        zipf.write(file)

# 이미지 파일 삭제 (옵션 - 현재 세션에서는 자동 삭제되므로 생략 가능)
